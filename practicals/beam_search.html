# beam search
def beam_search_decoder(data, k):
    sequences = [[list(), 0.0]]
    # walk over each step in sequence

    max_T, max_A = data.shape

    # Loop over time
    for t in range(max_T):
        all_candidates = list()
        # expand each current candidate
        for i in range(len(sequences)):
            seq, score = sequences[i]
            # Loop over possible alphabet outputs
            for c in range(max_A):
                candidate = [seq + [c], score - log(data[t, c])]
                all_candidates.append(candidate)
        # order all candidates by score
        ordered = sorted(all_candidates, key=lambda tup:tup[1])
        # select k best
        sequences = ordered[:k]
        
    # collapse repeated chars
    for seq in sequences:
        new_seq = []
        new_seq.append(seq[0][0])
        for i in range(1, len(seq[0])):
            if seq[0][i-1] != seq[0][i]:
                new_seq.append(seq[0][i])
        seq[0] = new_seq
        
    return sequences
